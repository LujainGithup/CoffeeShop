<script>
    const customerLink = document.getElementById('customerLink');
    const customerTable = document.getElementById('customerTable');

    let isCustomerTableVisible = false;

    customerLink.addEventListener('click', (event) => {
        event.preventDefault();
        console.log('Customer link was clicked.');

        if (isCustomerTableVisible) {
            // Hide the customer table
            customerTable.style.display = 'none';
            isCustomerTableVisible = false;
        } else {
            // Show the customer table
            customerTable.style.display = 'block';
            isCustomerTableVisible = true;
        }
    });
